<?php

// 1. Create Book class
class Book {
    // Properties
    private $title;
    private $author;
    private $year;
    
    // 2. Constructor to initialize properties
    public function __construct($title, $author, $year) {
        $this->title = $title;
        $this->author = $author;
        $this->year = $year;
    }
    
    // Method to get book details
    public function getDetails() {
        return "Title: {$this->title}, Author: {$this->author}, Publication Year: {$this->year}";
    }
    
    // Setter methods
    public function setTitle($title) {
        $this->title = $title;
    }
    
    public function setAuthor($author) {
        $this->author = $author;
    }
    
    public function setYear($year) {
        $this->year = $year;
    }
    
    // Getter methods (bonus)
    public function getTitle() {
        return $this->title;
    }
    
    public function getAuthor() {
        return $this->author;
    }
    
    public function getYear() {
        return $this->year;
    }
}


// 3. Create object and demonstrate functionality
echo "Creating Book object:\n\n";


// Create a book object
$book1 = new Book("To Kill a Mockingbird", "Harper Lee", 1960);


// Display initial details
echo "Initial Book Details:\n";
echo $book1->getDetails() . "\n\n";


// Update properties using setter methods
echo "Updating book properties...\n";
$book1->setTitle("1984");
$book1->setAuthor("George Orwell");
$book1->setYear(1949);


// Display updated details
echo "\nUpdated Book Details:\n";
echo $book1->getDetails() . "\n\n";


// Create another book object
$book2 = new Book("The Great Gatsby", "F. Scott Fitzgerald", 1925);
echo "Second Book:\n";
echo $book2->getDetails() . "\n\n";


// Demonstrate multiple objects
$book3 = new Book("Pride and Prejudice", "Jane Austen", 1813);
echo "Third Book:\n";
echo $book3->getDetails() . "\n";


?>
